% Captions and caption fonts
\usepackage{caption}
\DeclareCaptionFont{CaptionLabelFn}{\fontsize{10pt}{1em}\bodyfont\bfseries\color{darktext} #1\\}
\DeclareCaptionFont{CaptionTextFn}{\fontsize{9pt}{1em}\bodyfontlight\upshape\color{text} #1}
\DeclareCaptionLabelFormat{CaptionLabelFmt}{#1}
\captionsetup[figure]{
  font=CaptionTextFn,
  labelfont=CaptionLabelFn,
  labelformat=CaptionLabelFmt,
  labelsep=newline,
  justification=raggedright,
  format=plain,
  width=0.33\linewidth
}

% Figure command
\newcommand{\cvexample}[4]{
  \begin{figure}
    \renewcommand{\figurename}{#1}
    \centering
    \centerline{\includegraphics[keepaspectratio=true, interpolate, width=#3\linewidth]{#2}}
    \caption{#4}
    \label{fig:#1}
    \makecvfooter{\today}{Ryan Larson}{Example \#\thepage}
  \end{figure}
  \newpage
}
